class Celular

def __init__(self, marca, cor, modelo)
    self.marca = marca
    self.cor = cor
    self.modelo = modelo

def get_marca(self):
    return self.marca

def get_cor(self):
    return self.cor

def get_modelo(self):
    return self.modelo
